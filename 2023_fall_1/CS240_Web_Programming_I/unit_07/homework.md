// Javascript uses a MergeSort algorithm 
  function dataSort() {  
    //----- Reverse the order  
    for (let i=0; i<companies.length; i++) { 
      companies[i] = reverseAddress(companies[i]);  
    } 
    //----- Sort the array in ascending order - sorts based on the first column
    companies.sort();      
    //----- Reverse the order again.  
    for (let i=0; i<companies.length; i++) {
      companies[i] = reverseAddress(companies[i]);
    }
  } // end dataSort   
  // ======================================================================================
  // Reverses the order of the data list
  // Used the parameter oneCompany which represents the company name. 
  function reverseAddress(oneCompany) {
    var temp = oneCompany[0];       // Temp = Canvas = oneCompany[0]
    oneCompany[0] = oneCompany[1];  // FLIPS columns!!! Canvas=Construction
    oneCompany[1] = temp;           // Construction = Canvas
    return oneCompany;              // This returns entire row of data.
